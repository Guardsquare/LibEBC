280 template<class T>
281 bool UnlabeledValueArg<T>::processArg(int *i, std::vector<std::string>& args)
282 {
283 
284  if ( _alreadySet )
285  return false;
286 
287  if ( _hasBlanks( args[*i] ) )
288  return false;
289 
290  // never ignore an unlabeled arg
291 
292  _extractValue( args[*i] );
293  _alreadySet = true;
